## Columns and Rows

<Playground>
  <ThemeProvider>
    <Table
      name='TableRow'
      caption="Table caption"
      baseRoute='/table/#'
      hook={useTableRow}
      uidKey='_id'

    >
      <TableColumn heading='Full Name' />
      <TableColumn heading='User ID' />

      <TableRow uid='row'>
        {({data}) => {
          return <Fragment>
            <td>{`${data.givenName} ${data.familyName}`}</td>
            <td>{data._id}</td>
          </Fragment>
        }}
      </TableRow>

  </Table>

  </ThemeProvider>
</Playground>

<Playground>
  <ThemeProvider>
    <Table
      name='TableRows'
      caption="Table caption"
      baseRoute='/table/#'
      hook={useTableRow}
      uidKey='_id'
    >
      <TableColumn heading='Key' />
      <TableColumn heading='Value' />

      <TableRows uid='rows'>
        {({ data }) => {
          return Object.keys(data).map(key => <tr key={key}>
            <td>{key}</td>
            <td>{data[key]}</td>
          </tr>)
        }}
      </TableRows>
    </Table>

  </ThemeProvider>
</Playground>
